class Film {
  int _id;
  String _nama;
  String _emoji;
  double _rating;
  int _harga;

  Film(this._id, this._nama, this._emoji, this._rating, this._harga);

  int get id => _id;
  String get nama => _nama;
  String get emoji => _emoji;
  double get rating => _rating;
  int get harga => _harga;

  void tampilkanInfo() {
    print("$_id. $_emoji $_nama ⭐ $_rating - Rp $_harga");
  }
}

void main() {
  Film film1 = Film(1, "Avengers", "🦸", 4.8, 50000);
  film1.tampilkanInfo();
}

Class kursi

class Kursi {
  String kode;
  String status;
  int harga;

  Kursi(this.kode, this.status, this.harga);

  bool get isAvailable => status == "available";
  bool get isBooked => status == "booked";

  void pilih() {
    if (isAvailable) status = "selected";
  }

  void batalPilih() {
    if (status == "selected") status = "available";
  }
}

void main() {
  Kursi k1 = Kursi("A1", "available", 50000);

  print(k1.status); // available

  k1.pilih();
  print(k1.status); // selected

  k1.batalPilih();
  print(k1.status); // available
}

Class Pemesanan

class Pemesanan {
  String id;
  String film;
  String studio;
  String waktu;
  List kursi;
  int total;

  Pemesanan({
    required this.id,
    required this.film,
    required this.studio,
    required this.waktu,
    required this.kursi,
    required this.total,
  });

  void tampilkanDetail() {
    print("\n┌─────────────────────────────────────────┐");
    print("│ 🎟️ TIKET #${id.length > 6 ? id.substring(id.length - 6) : id}");
    print("│ Film    : $film");
    print("│ Studio  : $studio");
    print("│ Waktu   : $waktu");
    print("│ Kursi   : ${kursi.join(", ")}");
    print("│ Total   : Rp $total");
    print("└─────────────────────────────────────────┘");
  }
}

void main() {
  Pemesanan p = Pemesanan(
    id: "TRX123456",
    film: "Avengers",
    studio: "VIP",
    waktu: "19:00",
    kursi: ["A1", "A2"],
    total: 100000,
  );

  p.tampilkanDetail();
}

Class User

class Pemesanan {
  int total;

  Pemesanan(this.total);
}

class User {
  String email;
  List riwayatPemesanan = [];

  User(this.email);

  void tambahPemesanan(Pemesanan p) {
    riwayatPemesanan.add(p);
  }

  int getTotalBelanja() {
    int total = 0;
    for (var p in riwayatPemesanan) {
      total += p.total;
    }
    return total;
  }
}

void main() {
  User user = User("user@gmail.com");

  user.tambahPemesanan(Pemesanan(50000));
  user.tambahPemesanan(Pemesanan(75000));

  print(user.email);
  print(user.getTotalBelanja());
}
