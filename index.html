🧠 PENERAPAN PILAR OOP (OBJECT ORIENTED PROGRAMMING)

1. ENKAPSULASI

Atribut dipesan pada class Kursi bersifat privat dan hanya dapat diubah melalui method pesan() dan batalPesan(). Setiap class bertanggung jawab atas datanya sendiri tanpa bisa diakses langsung dari luar.

2. SINGLE RESPONSIBILITY

Setiap class memiliki 1 tanggung jawab: Film (data film), Studio (data studio), Kursi (status kursi), Pembeli (data pembeli), Tiket (data transaksi). Tidak ada class yang menangani banyak tugas sekaligus.

3. INTERAKSI OBJEK

Pembeli memiliki Tiket melalui relasi 1-to-many. Tiket menyimpan referensi ke Film, Studio, dan Kursi yang dipesan. Perubahan status kursi dilakukan melalui method pesan() dan batalPesan().

4. ABSTRACTION

Setiap class menyembunyikan detail internal dan hanya mengekspos method yang diperlukan. Pengguna class cukup memanggil info(), pesan(), beliTiket(), atau cetak() tanpa perlu tahu cara kerja di dalamnya.

// Algoritma Pemesanan Tiket CineBook
ALGORITMA PemesananTiket

DEKLARASI:
    pembeli: String
    film: Objek Film
    studio: Objek Studio
    jam: String
    kursiSelected: List dari String
    kursi: Objek Kursi
    hargaAwal, totalAkhir: Integer
    tiket: Objek Tiket

MULAI:
    BACA pembeli, film, studio, jam, kursiSelected

    JIKA film == NULL MAKA
        TAMPILKAN "Pilih film terlebih dahulu!"
        SELESAI
    AKHIR JIKA

    JIKA studio == NULL MAKA
        TAMPILKAN "Pilih studio terlebih dahulu!"
        SELESAI
    AKHIR JIKA

    JIKA PANJANG(kursiSelected) == 0 MAKA
        TAMPILKAN "Pilih minimal 1 kursi!"
        SELESAI
    AKHIR JIKA

    hargaAwal = 0

    UNTUK SETIAP kodeKursi DI kursiSelected LAKUKAN
        kursi = CARI_KURSI(kodeKursi)

        JIKA kursi.dipesan == TRUE MAKA
            TAMPILKAN "Kursi sudah dipesan: " + kodeKursi
        LAIN
            hargaAwal = hargaAwal + film.harga
            kursi.dipesan = TRUE
        AKHIR JIKA
    AKHIR UNTUK

    totalAkhir = hargaAwal * studio.multiplier

    tiket = BUAT_TIKET(
        pembeli,
        film,
        studio,
        jam,
        kursiSelected,
        totalAkhir
    )

    SIMPAN tiket KE tiketList

    PANGGIL renderTiket()
    PANGGIL switchTab("tiket")
    PANGGIL resetBooking()

SELESAI

class Film {
  final int id, harga, rating;
  final String judul, genre;

  const Film(this.id, this.judul, this.harga, this.rating, this.genre);
}

class Studio {
  final int id;
  final String nama;
  final double multiplier;

  const Studio(this.id, this.nama, this.multiplier);
}

class Kursi {
  final String kode;
  final Map status = {};

  Kursi(this.kode);

  bool isTersedia(String key) => status[key] != true;
  void pesan(String key) => status[key] = true;
}

class Pembeli {
  final String nama;
  final List tiketList = [];

  Pembeli(this.nama);

  void beli(Tiket t) => tiketList.add(t);
}

class Tiket {
  final String id;
  final String nama;
  final String kursi;
  final int total;

  const Tiket(this.id, this.nama, this.kursi, this.total);

  void cetak() {
    print("$id | $nama | Kursi $kursi | Rp$total");
  }
}

void main() {
  final film = [
    Film(1, "Avengers", 50000, 5, "Action"),
    Film(2, "Oppenheimer", 60000, 5, "Drama"),
    Film(3, "Inside Out", 45000, 4, "Animation"),
  ];

  final studio = [
    Studio(1, "Reguler", 1.0),
    Studio(2, "VIP", 1.5),
    Studio(3, "IMAX", 2.0),
  ];

  // ✔ FIX: tambah B1–B5 agar tidak error
  final kursi = [
    ...List.generate(6, (i) => Kursi("A${i + 1}")),
    ...List.generate(5, (i) => Kursi("B${i + 1}")),
  ];

  final pembeli = [
    Pembeli("Azriel"),
    Pembeli("Esti"),
    Pembeli("Ratu"),
    Pembeli("Alicia"),
    Pembeli("Budi"),
    Pembeli("Siti"),
    Pembeli("Andi"),
    Pembeli("Dewi"),
    Pembeli("Rizky"),
    Pembeli("Maya"),
  ];

  int counter = 1;

  // ================== 10 TRANSAKSI ==================
  final pesanan = [
    ["Azriel", 0, 0, "19:30", ["A1","A2"]],
    ["Esti",   0, 0, "19:30", ["A3"]],
    ["Ratu",   1, 1, "21:00", ["A4"]],
    ["Alicia", 1, 1, "21:00", ["A5"]],
    ["Budi",   0, 0, "19:30", ["A6"]],
    ["Siti",   1, 1, "21:00", ["B1"]],
    ["Andi",   2, 2, "15:00", ["B2"]],
    ["Dewi",   2, 2, "15:00", ["B3"]],
    ["Rizky",  0, 0, "19:30", ["B4"]],
    ["Maya",   1, 1, "21:00", ["B5"]],
  ];

  for (final p in pesanan) {
    final nama = p[0] as String;
    final f = film[p[1] as int];
    final s = studio[p[2] as int];
    final jam = p[3] as String;
    final listKursi = (p[4] as List).cast();

    final pemb = pembeli.firstWhere((x) => x.nama == nama);

    for (final k in listKursi) {
      final kursiObj = kursi.firstWhere((x) => x.kode == k);

      if (kursiObj.isTersedia(jam)) {
        kursiObj.pesan(jam);

        final total = (f.harga * s.multiplier).round();

        final tiket = Tiket(
          "TIX${counter.toString().padLeft(3, '0')}",
          nama,
          k,
          total,
        );

        pemb.beli(tiket);

        print("✔ $nama -> ${f.judul} | $k | Rp$total");
        counter++;
      }
    }
  }
}
